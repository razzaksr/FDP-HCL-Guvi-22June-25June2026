view dbs:

	use DATABASE;
view collections:
	show collections
create/insert document into the collections:
	db.COLLECTIONNAME.insertOne({})
view documents in the collection:
	db.COLLECTIONNAME.find()
filter based atomic operators:
	db.COLLECTIONNAME.find({filter})
	eg:
	db.professor.find({"experience":{"$gte":15}})
	db.professor.find({"department":"ISE"})
update documents:
	db.COLLECTIONNAME.updateOne({where},{set})
	eg:
	db.professor.updateMany({"department":"ISE"},{$set:{"package":25.6}})
	db.professor.updateOne({"nane":"Razak Mohamed"},{$set:{"department":"ISE"}})
deletion document:
	db.COLLECTIONNAME.deleteOne/deleteMany
	eg:
	db.professor.deleteMany({"department":{$nin:["CSE","ISE"]}})
	db.professor.deleteOne({"package":{"$lte":10.0}})
	db.professor.deleteOne({"nane":"Razak Mohamed"})
